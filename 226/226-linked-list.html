function insertNodeAtPosition(head, data, position) {

var curr = head;

var prev = null;

var node = new SinglyLinkedListNode();

node.data = data;

 

var i = 0;

while (i < position) {

prev = curr;

curr = curr.next;

if (curr == null) { break;}

i++;

}

node.next = curr;

prev.next = node;

return head;

}
